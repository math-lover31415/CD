\documentclass[12pt]{article}

\usepackage{pdfpages}
\usepackage{ifthen}

\makeatletter
\newcommand{\getpdfpages}[2]{%
  \pdfximage{#1}%
  \edef#2{\the\pdflastximagepages}%
}

% Insert a page from blank.pdf as the "blank"
\newcommand{\blankpage}{%
  \includepdf[pages=1,pagecommand={\thispagestyle{empty}}]{blank.pdf}%
}

% \experiment{experiment.pdf}{output.pdf}
\newcommand{\experiment}[2]{%
  \getpdfpages{#1}{\pa}%
  \getpdfpages{#2}{\pb}%
  \newcount\countpa \countpa=\pa
  \newcount\countpb \countpb=\pb
  \newcount\N
  \ifnum\countpa>\countpb \N=\countpa \else \N=\countpb \fi
  \newcount\ii \ii=1
  \loop
    % odd page -> output
    \ifnum\ii>\countpb
      \blankpage
    \else
      \includepdf[pages={\the\ii},pagecommand={\thispagestyle{empty}}]{#2}
    \fi
    % even page -> experiment
    \ifnum\ii>\countpa
      \blankpage
    \else
      \includepdf[pages={\the\ii},pagecommand={\thispagestyle{empty}}]{#1}
    \fi
    \advance\ii by 1
  \ifnum\ii>\N\else\repeat
}
\makeatother
